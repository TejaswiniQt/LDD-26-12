#include<linux/init.h>
#include<linux/module.h>

static int __init simple_hello_world_init(void)
{
	pr_info("Hello world\n");
	return 0;
}

void __exit simple_hello_world_exit(void)
{
	pr_info("Clean up function called\n");
}

module_init(simple_hello_world_init);
module_exit(simple_hello_world_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Tejaswini");
